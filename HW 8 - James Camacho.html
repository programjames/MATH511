(1)¶

We want to solve the following BVP: $$\epsilon\cdot\frac{d^2u(x)}{dx^2} + u(x)\left(\frac{du(x)}{dx}-1\right) = 0, u(a) = \alpha, u(b) = \beta$$

Geometric Mapping¶

Suppose we have some geometric mapping $x(\xi)$, with $x(0) = \alpha, x(1) = b$, and $\xi$ some uniformly spaced mesh. Let $\phi = x^{-1}$, and $u = v\circ \phi$, so we have

$$u' = v'\phi',$$$$u'' = v''(\phi')^2 + v'\phi''.$$

The equation becomes

$$\epsilon (v''(\phi')^2 + v'\phi'') + v(v'\phi'-1) = 0, v(0) = \alpha, v(1) = \beta$$

We use the finite difference approximations

$$v'' \approx \frac{v_{n-1} - 2v_{n} + v_{n+1}}{h^2},$$$$v'\approx \frac{v_{n+1}-v_{n-1}}{2h}$$

which are order $h^2$. We also need to calculate $\phi'$ and $\phi''$. We could fit to a high degree polynomial, but what's probably best is to use a spline.

Spline¶

We use the Chebyshev polynomials ($T_i$) as a basis for the spline. Suppose we have degree $d$ spline, with $k$ points in each piece, and a total of $p$ pieces. The $(d-k+1)$th derivative should be continuous across pieces. Let

$$F_n = \sum_{i=0}^{d}T_ic_{ni}$$

be the function within each spline.

Actually, this spline stuff is going to take a really long time to implement, so I'll just use scipy's spline interpolation.

Newton's Method¶

We need to calculate the Jacobian. We have

$$J_{i, i} = -\frac{2\epsilon(\phi')^2}{h^2} + (v'-1),$$$$J_{i, i-1} = \epsilon\cdot \left[\frac{(\phi')^2}{h^2} - \frac{\phi''}{2h}\right] - \frac{v\phi'}{2h},$$$$J_{i, i+1} = \epsilon\cdot \left[\frac{(\phi')^2}{h^2} + \frac{\phi''}{2h}\right] + \frac{v\phi'}{2h},$$

and zero everywhere else.

We code all of this below.

def dxi(f, n, degree=1):
    if degree == 1:
        return (f[2:] - f[:-2]) / 2 * n
    elif degree == 2:
        return (f[2:] + f[:-2] - 2*f[1:-1]) * n**2
    else:
        raise Exception("Invalid degree. Only 1 and 2 implemented right now.")
        
def Jacobian(v, dphi, ddphi, eps):
    """
    v is defined above in the math.
    dphi = phi', should only be phi'(x[1:-1]).
    ddphi = phi''
    eps = epsilon
    
    Returns: A banded matrix representing the Jacobian.
    """
    # Bdry points: x[0], x[-1]
    n = len(x) - 1
    
    d_0 = -2*eps * dphi**2 * n**2 + dxi(v, n) - 1
    d_m1 = eps * (dphi**2 * n**2 - ddphi/2*n) - v[1:-1]*dphi/2*n
    d_p1 = eps * (dphi**2 * n**2 + ddphi/2*n) + v[1:-1]*dphi/2*n
    
    d_0 = np.concatenate([[1], d_0, [1]])
    d_m1 = np.concatenate([d_m1, [0, 0]])
    d_p1 = np.concatenate([[0, 0], d_p1])
    
    return [d_p1, d_0, d_m1]

def Energy(v, dphi, ddphi, eps):
    n = len(v) - 1
    e = np.zeros_like(v)
    dv = dxi(v, n, 1)
    ddv = dxi(v, n, 2)
    e[1:-1] = eps * (ddv * dphi**2 + dv * ddphi) + v[1:-1]*(dv*dphi - 1)
    
    return e

def Newton(G, J, x, phi, v=None, maxiters=100, error=1e-10, **kwargs):
    """
    G - Function to minimize
    J - Jacobian of the function
    x - x positions we're evaluating the function at.
    phi - geometric mapping function. Should have a derivative function.
    v - Initial v guess.
    maxiters - Maximum number of iterations of Newton's method.
    kwargs is problem specific. In this case, it should have eps, alpha, and beta keywords
    """
    if v is None:
        v = np.zeros_like(x)
        
    eps = kwargs["eps"]
    v[0] = kwargs["alpha"]
    v[-1] = kwargs["beta"]
    
    dphi = phi.derivative(1)(x[1:-1])
    ddphi = phi.derivative(2)(x[1:-1])
    
    for i in range(maxiters):
        G_ = G(v, dphi, ddphi, eps)
        J_ = J(v, dphi, ddphi, eps)
        delta = solve_banded((1,1), J_, G_)
        
        if(np.linalg.norm(delta) <= np.linalg.norm(v) * error):
            break
        
        v -= delta
        v[0] = kwargs["alpha"]
        v[-1] = kwargs["beta"]
        
    return v

(3)¶

We want more points near the extremes of the second derivative of $u$. Here's how we do it: We interpolate $u$ with a spline, and find the function $w = |u''|$. We could just take the integral of $w$ to find a new geometric mapping, but then it won't be super smooth. So we convolve it with a gaussian first (similar idea to Green's functions, but using a Gaussian probability curve rather than the Dirac delta function). This gives our map

$$x(\xi) = \int_{0}^{\xi} (G * w)(t) \text{d}t = G * \int_{0}^x w(t)\text{d}t.$$

We won't be able to calculate $\phi$ exactly. We can use Newton-Raphson iteration to get a good approximation for $\phi'$ and $\phi''$, but I'm just interpolating with a spline.

It ends up working a little better, getting down to $2.5\cdot 10^{-3}$.

from scipy.signal import convolve
from scipy.integrate import cumulative_trapezoid

def update_mesh(x, phi, v, points=101):
    t = np.linspace(0, 1, len(x))
    u = Spline(x, v)
    w = abs(u.derivative(2)(t))
    sigma = 0.25
    G = 1 / sigma / (2*np.pi)**0.5 * np.exp(-0.5*((t-0.5)/sigma)**2)
    
    C = convolve(G, w, mode="same")
#     C += np.mean(C)*1e-2
    P = cumulative_trapezoid(C)
    P = np.sort(P)
    P = (P - P[0]) / (P[-1] - P[0])
    
    x = Spline(P, np.linspace(0, 1, len(P)))
    xi = np.linspace(0, 1, points)
    x = x(xi)
    x = np.sort(x)
    
    phi = Spline(x, xi, k=2, s=1)
    
    return x, phi

import numpy as np
from scipy.special import factorial
from scipy.optimize import fsolve

def get_c(k, d=2):
    """
    k - An array with distances from the center point for the finite difference approximation.
    d - The order of the derivatives (i.e. 2 => u'').
    """
    n = len(k)
    M = np.zeros((n, n))
    for i in range(n):
        M[i] = np.power(k, i)
    
    D = np.zeros(n)
    D[d] = factorial(d)
    
    return np.linalg.solve(M, D)

def get_row(i, mesh, order=2, d=2):
    """
    i - row index of the matrix
    mesh - the locations of the evaluation points
    order - order of the finite difference method
    d - derivative order
    """
    
    n = len(mesh)
    l = d + order
    if 2*i < l - 1:
        indices = range(l)
    elif 2*(n-i) < l-1:
        indices = range(n+1-l, n+1)
    else:
        indices = range(i-l//2, i+l//2)
        
    k = [mesh[ii] - mesh[i] for ii in indices]
    return indices, get_c(k, d=d)

def difference_index(i, mesh, u, order=None, d=None):
    if order is None:
        order = 2
    if d is None:
        d = 2
    
    cols, c = get_row(i, mesh, order=order, d=d)
    return u[cols] @ c

di = difference_index

def difference_prime(i, j, mesh, u, order=None, d=None):
    if order is None:
        order = 2
    if d is None:
        d = 2
        
    cols, c = get_row(i, mesh, order=order, d=d)
    if j in cols:
        return c[cols.index(j)]
    else:
        return 0
    
dp = difference_prime
    

def Jacobian(u, x, eps):
    n = len(x)
    J = np.zeros((n, n))
    
    # Bdry cond. (Dirichlet)
    J[0, 0] = 1
    J[-1, -1] = 1
    
    # Interior points
    for i in range(1, n-1):
        for j in range(1, n-1):
            J[i, j] = eps * dp(i, j, x, u, d=2)
            J[i, j] += u[i] * dp(i, j, x, u, d=1)
            if i == j:
                J[i, j] += dp(i, j, x, u, d=0) * (di(i, x, u, d=1) - 1)
    return J

def G(u, x, eps):
    n = len(x)
    g = np.zeros(n)
    # Bdry are zero, because it should be the exact solution already.
    # Interior points
    for i in range(1, n-1):
        g[i] = eps * di(i, x, u, d=2) + di(i, x, u, d=0) * (di(i, x, u, d=1) - 1)
    return g
    

def Newton(G, J, x, u=None, maxiters=20, error=1e-10, **kwargs):
    """
    G - Function to minimize
    J - Jacobian of the function
    x - x positions we're evaluating the function at.
    u - Initial u guess.
    maxiters - Maximum number of iterations of Newton's method.
    kwargs is problem specific. In this case, it should have eps, alpha, and beta keywords
    """
    if u is None:
        u = np.zeros_like(x)
        
    eps = kwargs["eps"]
    u[0] = kwargs["alpha"]
    u[-1] = kwargs["beta"]
    
    for i in range(maxiters):
        J_ = J(u, x, eps)
        G_ = G(u, x, eps)
        delta = np.linalg.solve(J_, G_)
        
        if(np.linalg.norm(delta) <= np.linalg.norm(u) * error):
            break
        
        u -= delta
        u[0] = kwargs["alpha"]
        u[-1] = kwargs["beta"]
            
    print(f"Took {i} iterations.")
        
    return u

def G2(u, x, eps, alpha, beta):
    n = len(x)
    g = np.zeros(n)
    # Bdry are zero, because it should be the exact solution already.
    # Interior points
    u[0] = alpha
    u[-1] = beta
    for i in range(1, n-1):
        g[i] = eps * di(i, x, u, d=2) + di(i, x, u, d=0) * (di(i, x, u, d=1) - 1)
    return g

def update_mesh(mesh, u, points=5):
    # Add points to the places with the highest slopes.
    indices = range(len(mesh)-1)
    
    # sort indices by the absolute value of the slopes
    indices = list(zip(*sorted(zip(abs((u[1:]-u[:-1])/(mesh[1:]-mesh[:-1])), indices), reverse=True)))[1]
    indices = np.array(indices)
    
    for p in range(points):
        i = indices[p]
        mesh = np.insert(mesh, i+1, (mesh[i]+mesh[i+1])/2)
        u = np.insert(u, i+1, (u[i]+u[i+1])/2)
        indices[indices>i] += 1
    return mesh, u
